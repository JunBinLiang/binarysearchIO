import java.util.*;

class Solution {
    public int solve(int n) {
        if(n==1)return 1;
        long dp[]=new long[n+1];
        dp[1]=1;
        dp[2]=2;
        for(int i=3;i<dp.length;i++){
            dp[i]=dp[i-1]+dp[i-2];
            dp[i]%=(1e9+7);
        }
        return (int)(dp[n]);
    }
}
